nmdrTable

is a pure Javascript extension of the HTML table that provides full customizable solution for representing tabular data on the web. It is part of the nmdrUI framework and supports sorting, filtering and pagination. Each cell of header, footer and body is customizable with own rendering. The data of table can be loaded dynamically through Ajax calls, so any server-side technology including PHP, ASP, Perl, Java Servlets and JSP can be used. Take a look at examples below for a better insight.

var tb = nmdr.table('div-C');
tb.getColumnWidth = function(col) { return 120; };
tb.build(
{
	dummyCols: 5,
	dummyRows: 500,
	showHeader: true,
	showFooter: true,
	showShadow: true,
	showRowCommands: true,
	menuColumn: 2,
	rowLimit: 50,
	sortingColumn: 3,
	sortingDirection: 'SortZA',
	showGrids: true,
	scrollable: true,
});
	

 No CSS needed
	

 A div element with id = 'div-C'
	

var tb = nmdr.table('div-D');

tb.prepareData = function () {

	var rowdata = [], s = this.props.dataSplitter;
	for (var i = 0; i < 400; i++) {
		rowdata[i] = "Name " + i + s + "Adress " + i + s + "Email " + i + s + "Phone " + i;
	}

	return { columns: ["Name", "Adress", "Email", "Phone"], rows: rowdata };
};

tb.build(
{
	showHeader: true,
	showFooter: true,
	showShadow: true,
	showRowCommands: true,
	multiSelection: true,
	menuColumn: 2,
	sortingColumn: 3,
	sortingDirection: tb.SORT_ASCENDING,
	filters: { 3: 'contains&0', 4: 'contains&1' }
});
	

 No CSS needed
